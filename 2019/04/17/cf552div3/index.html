A. Restoring Three Numbers

题意：给你四个数，a+b，a+c，b+c，a+b+c，（abc都为正数，可以相等）让你找出满足条件的a、b、c的值，排名不分先后

解答：用带我入坑的大佬的话说就是“人尽皆知傻逼题”
找出四个数中最大的数，减去其他三个就ok了了
#include <bits/stdc++.h> using namespace std; int main() { int a[6]; int maxn = 0,flag = 0; for(int i = 1;i <= 4;i++) { scanf("%d",&a[i]); if(a[i] > maxn) { maxn = a[i];flag = i; } } int cnt = 0; for(int i = 1;i<=4;i++) { if(i!=flag) { printf("%d",maxn-a[i]); cnt++; if(cnt<=2) printf(" "); } } printf("\n"); return 0; }

B. Make Them Equal

题意：给定一个序列，让其中的每一个数通过+x，-x，或不变三种方式变成同一个数，求最小的非负数x，不存在输出-1

解决：数列中超过三种数肯定是错的，所以先用set求不同数的个数
如果只有一种数，每个数不变就好，答案为0；
两种，有两个情况，他们有中位数，直接中间的数减去小的数，没有就只能是两个数的差值；
三种，只有等差数列才满足条件
#include <bits/stdc++.h> using namespace std; const int maxn = 105; int main() { int n,tem; int num[5]; scanf("%d",&n); int a = 0,b = 0,c = 0; set<int> s; set<int> ::iterator it; for(int i = 1;i <= n;i++) { scanf("%d",&tem); s.insert(tem); } int x = s.size(); if(x>3) printf("-1\n"); else if(x == 1) printf("0\n"); else { int i = 1; for(it=s.begin();it!=s.end();it++) { num[i++] = *it; } if(x == 2) { int sum = num[1]+num[2]; if(sum%2==0) printf("%d\n",sum/2-num[1]); else printf("%d\n",num[2]-num[1]); } else { int sum = num[1]+num[3]; if(sum == num[2]*2) printf("%d",num[2]-num[1]); else printf("-1\n"); } } return 0; }

C. Gourmet Cat

这一题当时思索了一下，先跳过去写D，然后再重新思考，交题的时候刚好断网，换卡开热点惨兮兮
题意：一只猫从星期一到星期天每天固定只吃1 2 3 1 3 2 1类食物，三类食物分别有a、b、c数量，主人想出去旅游，从星期几开始无所谓，问这只贵气的猫最多能撑多久

解答：数据比较大啊……(1≤a,b,c≤7⋅10^8)不能纯暴力。可以看出每周1、2、3类食物消耗的量分别是3、2、2，于是我么你先用a、b、c处以3、2、2，看看哪种食物最不能撑
然后分别多留出一周，分三种情况暴力处理从一周每一天开始的情况，求出坚持最大天数才使三类存粮最慢消耗为负数
#include <bits/stdc++.h> using namespace std; const int maxn = 200005; int main() { int n,a,b,c; int s[8][15]={ {1,2,3,1,3,2,1,1,2,3,1,3,2,1}, {2,3,1,3,2,1,1,2,3,1,3,2,1,1}, {3,1,3,2,1,1,2,3,1,3,2,1,1,2}, {1,3,2,1,1,2,3,1,3,2,1,1,2,3}, {3,2,1,1,2,3,1,3,2,1,1,2,3,1}, {2,1,1,2,3,1,3,2,1,1,2,3,1,3}, {1,1,2,3,1,3,2,1,1,2,3,1,3,2} }; scanf("%d %d %d",&a,&b,&c); int x,y,z; x = a/3; y = b/2; z = c/2; int ans = 0; if(x<=y&&x<=z) { x--; ans+=x*7; a-=3*x;b-=2*x;c-=2*x; int ma=0; for(int i=0;i<7;i++) { int cnt=0; x=a,y=b,z=c; for(int j=0;j<14;j++) { if(s[i][j]==1) x--; else if(s[i][j]==2) y--; else if(s[i][j]==3) z--; if(y<0||x<0||z<0) break; cnt++; } ma=max(ma,cnt); } ans+=ma; } else if(y<x&&y<z) { y--; ans+=y*7; a-=3*y;b-=2*y;c-=2*y; int ma=0; for(int i=0;i<7;i++) { int cnt=0; x=a,y=b,z=c; for(int j=0;j<14;j++) { if(s[i][j]==1) x--; else if(s[i][j]==2) y--; else if(s[i][j]==3) z--; if(y<0||x<0||z<0) break; cnt++; } ma=max(ma,cnt); } ans+=ma; } else { z--; ans+=z*7; a-=3*z;b-=2*z;c-=2*z; int ma=0; for(int i=0;i<7;i++) { int cnt=0; x=a,y=b,z=c; for(int j=0;j<14;j++) { if(s[i][j]==1) x--; else if(s[i][j]==2) y--; else if(s[i][j]==3) z--; if(y<0||x<0||z<0) break; cnt++; } ma=max(ma,cnt); } ans+=ma; } printf("%d\n",ans); return 0; }

D. Walking Robot

题意：一个机器人要从0走到n，它拥有b节普通电池和a节太阳能蓄力电池，每走一步路需要消耗一节电池，这段路上另外有个数组si表示i-1到i处是否有太阳，为1表示有太阳，有阳光的话可以给不在使用的太阳能蓄力电池充电，求小机器人最多能走几步（n封顶）

解答:一遍模拟，精髓就在于一定要先用太阳能电池，其他没啥了
#include <bits/stdc++.h> using namespace std; const int maxn = 200005; int main() { int n,a,b,x,y; int s[maxn]; scanf("%d %d %d",&n,&a,&b); x = a; y = b; for(int i = 1;i <= n;i++) { scanf("%d",&s[i]); } int flag = -1; for(int i = 0;i < n; i++) { if(y<b&&x>0&&s[i+1]==1) { y++;x--; } else if(y>0) { y--; } else if(x>0) { x--; } else { flag = i;break; } } if(flag == -1) printf("%d\n",n); else printf("%d\n",flag); return 0; }
